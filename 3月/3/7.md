### 二叉树 递归 数学归纳法 栈
#### 104.二叉树的最大深度

方法一：递归 自底向上
求最大深度，就是求左右子树的最大深度，最后返回其中的最大值
临界点就是子树为空节点的时候返回0
注意：空间复杂度为o(n)，因为递归的时候调用了栈保存节点信息，当二叉树全是左子树或者右子树的时候，空间复杂度就是o(n)
```cpp
class Solution {
public:
    int maxDepth(TreeNode* root) {
        if(root==nullptr){//临界
            return 0;
        }
        int left=maxDepth(root->left);
        int right=maxDepth(root->right);
        return max(left,right)+1;
    }
};

```
方法二：维护一个全局变量ans，然后递归遍历整个二叉树 ，自顶向下

```cpp
class Solution {
public:
    int maxDepth(TreeNode* root) {
        int ans=0;
        //1. this：表明这是 Lambda 的自引用参数；
        // 2. auto&& dfs：声明一个万能引用的参数 dfs，指向 Lambda 自身；
        // 3. 有了这个参数，Lambda 内部就能通过 dfs(...) 递归调用自己。
        auto dfs=[&](this auto && dfs,TreeNode* root, int cnt)->void{ //这里用到了自参数
            if(root==nullptr){
                return ;
            }
            cnt++;
            ans=max(cnt,ans);
            dfs(root->left,cnt);
            dfs(root->right,cnt);
        };
        dfs(root,0);
        return ans;
    }
};
```

#### 111.二叉树的最小深度
为什么最大深度的想法不能用？因为node一边为null，一边为树的时候，那很显然最小深度是null的那边，但node并不是叶子节点
我们需要先判断当前节点是否为叶子节点

怎么判断是否为叶子节点呢？

方法一：自顶向下
```cpp
class Solution {
public:
    int minDepth(TreeNode* root) {
        int ans=INT_MAX;
        auto dfs=[&](this auto && dfs,TreeNode *Node,int depth){
            if(Node==nullptr){
                return ;
            }
            depth++;
            if(Node->right==nullptr&&Node->left==nullptr){
                 ans=std::min(ans,depth);//只有当前节点为叶子节点的时候才更新min值
            }
            dfs(Node->left,depth);
            dfs(Node->right,depth);
        };
        dfs(root,0);
        return root?ans:0;
    }
};
```
方法二：自底向上
```cpp
class Solution {
public:
    int minDepth(TreeNode* root) {
        if(root==nullptr){//如果 node 是空节点，由于没有节点，返回 0
            return 0;
        }
        if(root->left==nullptr){//如果 node 没有右儿子，那么深度就是左子树的深度加一
            return minDepth(root->right)+1;
        }
        if(root->right==nullptr){//如果 node 没有左儿子，那么深度就是右子树的深度加一
            return minDepth(root->left)+1;
        }
        //如果 node 左右儿子都有，那么分别递归计算左子树的深度，以及右子树的深度，二者取最小值再加一
        return min(minDepth(root->right),minDepth(root->left))+1;
    }
};
```

方法三：用求最大深度的方法，但是加上一个判断空子树的情况
 ```cpp
 int minDepth(TreeNode* root) {
        if(root==nullptr){//如果 root 是空节点，由于没有节点，返回 0
            return 0;
        }
        int left=minDepth(root->left);
        int right=minDepth(root->right);
        //已经知道左右子树的最小值，那直接判断是否为0，如果有一个为0，意味着有一边是空的，一边是子树，那取他们的最大值，如果都是0，那就是返回0，如果都不是0，那就是说明左右子树都在，继续递归
        int a=0;
        if(left==0||right===0){
            a=max(left,right);
        }else{
            a=min(left,right);
        }
        return a+1;
    }
```

#### 404 左叶子之和
咋每个题目都不会啊！！！


//自底向上，当前节点的答案 = 左子树答案 + 右子树答案 + 当前节点的贡献
```cpp
class Solution {
public:
    int sumOfLeftLeaves(TreeNode* root) {
        if(root==nullptr){//空树
            return 0;
        }
        int sum= sumOfLeftLeaves(root->left) + sumOfLeftLeaves(root->right);
        TreeNode *l=root->left;
        if(l&&l->right==nullptr&&l->left==nullptr){
            sum+= l->val;
        }
        return sum;
    }
};
```
方法二：自顶向下，遍历整棵树，收集所有符合条件的节点值
```cpp
class Solution {
public:
    int sumOfLeftLeaves(TreeNode* root) {
        int sum=0;
        auto dfs=[&](this auto && dfs,TreeNode* node){
            if(node==nullptr){
                return;
            }
            dfs(node->left);
            dfs(node->right);
            TreeNode *left=node->left;
            if(left!=nullptr&&left->right==nullptr&&left->left==nullptr){
                sum+=left->val;
            }
        };
        dfs(root);
        return sum;
    }
};
```

#### 112.路径总和
终于自己解决了一个

方法一：自底向上
```cpp
class Solution {
public:
    bool hasPathSum(TreeNode* root, int targetSum) {
        if(root==nullptr){
            return false;
        }
        int sum=targetSum-root->val;
        if(sum==0&&root->left==nullptr&&root->right==nullptr){
            return true;
        }
        bool right=hasPathSum(root->right,sum);
        bool left=hasPathSum(root->left,sum);
        return right||left;
    }
};

```

自顶向下

```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    bool hasPathSum(TreeNode* root, int targetSum) {
        bool ans=false;
        auto dfs=[&](this auto && dfs,TreeNode *node,int sum){
            if(!node){
                return;
            }
            sum+=node->val;
            if(!node->left&&!node->right&&sum==targetSum){
                ans=true;
            }
            dfs(node->left,sum);
            dfs(node->right,sum);
        };
        dfs(root,0);
        return ans;
    }
};
```


#### 129.求根节点到叶子节点数字之和
自解决

```cpp
//自顶向下
class Solution {
public:
    int sumNumbers(TreeNode* root) {
        int sum=0;
        auto dfs=[&](this auto && dfs,TreeNode *node,int s){
            if(!node){
                return;
            }
            s+=node->val;//加上当前节点的值
            if(!node->right&&!node->left){//如果是叶子节点，那就加到答案里面
                sum+=s;
                return;
            }else{//如果不是叶子节点，说明下面还有节点
                s*=10;
            }
            dfs(node->left,s);
            dfs(node->right,s);
        };
        dfs(root,0);
        return sum;
    }
};
```

//自底向上,这里必须在递归函数中加入一个参数来传递当前节点的和，其实还是自顶向下的感觉
```cpp
class Solution {
public:
    int sumNumbers(TreeNode* root,int x=0) {
        if(!root){
            return 0;
        }
        x=x*10+root->val;
        if(!root->left&&!root->right){
            return x;//这里必须直接返回，否则继续递归的话会把这条路径的和设为0
        }
        return sumNumbers(root->left,x)+sumNumbers(root->right,x);
    }
};
```
